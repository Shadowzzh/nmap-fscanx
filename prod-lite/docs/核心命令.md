cd /tmp/fscanx-manual/phase1

fscanx \
  -h '192.168.1.0/24,192.168.20.0/24,192.168.4.0/24' \
  -p 22,80,443,445,3389 \
  -np \
  -t 4000 \
  -time 1 \
  -nocolor \
  -json

说明：

- `-json` 只会把输出内容改成 JSON 片段
- 默认结果文件仍然是当前工作目录下的 `result.txt`
- `-json` 不会自动生成 `result.json`

中间提取 alive_ips.txt：

cd /tmp/fscanx-manual/phase1

awk '
  NF {
    line = $0
    sub(/,[[:space:]]*$/, "", line)
    lines[++count] = line
  }
  END {
    print "["
    for (i = 1; i <= count; i++) {
      suffix = ""
      if (i < count) {
        suffix = ","
      }
      print lines[i] suffix
    }
    print "]"
  }
' result.txt > normalized.json

jq -r '
  .[]
  | select(.type == "Port")
  | try (.text | capture("open\\t(?<ip>[0-9.]+):(?<port>[0-9]+)").ip) catch empty
' normalized.json | awk 'NF && !seen[$0]++' > alive_ips.txt
